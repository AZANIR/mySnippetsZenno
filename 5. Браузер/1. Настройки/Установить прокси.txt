//Instance.SetProxy - установка прокси сервера для работы текущего инстанса
instance.SetProxy("127.0.0.1", 8080, "socks5", "user", "password", false); //Первая перегрузка метода
instance.SetProxy("socks5://user:password@127.0.0.1:8080", false); //Вторая перегрузка метода

//Примечания:
//1. Параметры первой перегрузки метода SetProxy (https://help.zennolab.com/en/v5/zennoposter/5.9.9/webframe.html#topic216.html):
//	первый - адрес прокси сервера
//	второй - порт прокси-сервера
//	третий - протокол прокси сервера (http, socks4 или socks5)
//	четвёртый - пользовательпрокси-сервера
//	пятый - пароль прокси сервера
//	шестой - использовать ли proxifier при установке соединения
//2. Строковой параметр второй перегрузки метода должен быть в формате: [protocol://][(login):(password)@](ip):(port)
